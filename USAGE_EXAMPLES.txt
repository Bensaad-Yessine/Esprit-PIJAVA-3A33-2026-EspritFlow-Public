// EXAMPLE 1: PASSWORD RESET EMAIL (Mot de passe oublié)
// ─────────────────────────────────────────────────────────────────────────────

// When user clicks "Forgot password?" in login screen:

PasswordResetService passwordResetService = new PasswordResetService();

try {
    // This automatically:
    // 1. Creates a reset request in database
    // 2. Generates a unique token
    // 3. Creates beautiful HTML email
    // 4. Sends email via SMTP
    passwordResetService.requestReset("user@example.com");

    // User receives email with:
    // - Unique reset code (30 min validity)
    // - Step-by-step instructions
    // - Security notice
    // - Support contact

} catch (Exception e) {
    System.err.println("Failed to request password reset: " + e.getMessage());
}

// Then user enters the code and new password:

try {
    passwordResetService.resetPassword(
        "user@example.com",
        "abc123.def456ghi789",  // reset code from email
        "newSecurePassword123"    // new password
    );
    // Password updated successfully!
} catch (Exception e) {
    System.err.println("Failed to reset password: " + e.getMessage());
}


// EXAMPLE 2: BAN NOTIFICATION EMAIL (Notification de bannissement)
// ─────────────────────────────────────────────────────────────────────────────

// When administrator bans a user:

UserServices userServices = new UserServices();

// This automatically:
// 1. Marks user as banned in database
// 2. Creates ban notification HTML email
// 3. Sends email with:
//    - Red alert header
//    - Ban reason
//    - Consequences
//    - Appeal instructions
//    - Support contact

userServices.banUser(
    42,                                    // user ID
    "Spam et harcèlement detectes"        // ban reason
);

// User receives email with the reason and instructions on how to appeal


// EXAMPLE 3: UNBAN NOTIFICATION EMAIL (Notification de déblocage)
// ─────────────────────────────────────────────────────────────────────────────

// When administrator unbans a user:

UserServices userServices = new UserServices();

// This automatically:
// 1. Removes ban flag from database
// 2. Creates unban success email
// 3. Sends email with:
//    - Green success header
//    - Confirmation message
//    - List of restored access
//    - Call-to-action button
//    - Support contact

userServices.unbanUser(42);  // user ID

// User receives email confirming account is reactivated

// BAN/UNBAN IN ADMIN CONTROLLER (example):

void banUserFromAdmin(int userId, String reason) {
    UserServices userServices = new UserServices();
    userServices.banUser(userId, reason);  // Email sent automatically!
    showAlert("User banned and notification email sent");
}

void unbanUserFromAdmin(int userId) {
    UserServices userServices = new UserServices();
    userServices.unbanUser(userId);  // Email sent automatically!
    showAlert("User unbanned and confirmation email sent");
}

// ═══════════════════════════════════════════════════════════════════════════════
// ERROR HANDLING
// ═══════════════════════════════════════════════════════════════════════════════

try {
    passwordResetService.requestReset("user@example.com");
} catch (IllegalArgumentException e) {
    // Email invalid or user not found
    System.out.println("Error: " + e.getMessage());
    // Show user-friendly error message
} catch (java.io.IOException e) {
    // SMTP error - credentials wrong, server down, etc.
    System.out.println("Email sending failed: " + e.getMessage());
    // Show error: "Failed to send email, try again later"
} catch (SQLException e) {
    // Database error
    System.out.println("Database error: " + e.getMessage());
}
